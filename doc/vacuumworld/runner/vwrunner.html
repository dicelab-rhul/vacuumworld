151    def __validate_minds(self) -> None:
152        try:
153            if not all(isinstance(self.__minds[colour], self.__allowed_args[str(colour) + "_mind"]) for colour in VWColour if colour != VWColour.user):
154                allowed_minds: dict[str, type] = {str(colour): self.__allowed_args[str(colour) + "_mind"] for colour in VWColour if colour != VWColour.user}
155
156                raise TypeError(f"One or more mind surrogates are not subclasses of the allowed types: {allowed_minds}.")
157
158            for colour, mind in self.__minds.items():
159                if colour != VWColour.user:
160                    VWActorMindSurrogate.validate(mind=mind, colour=colour, surrogate_mind_type=self.__allowed_args[str(colour) + "_mind"])
161                else:
162                    VWActorMindSurrogate.validate(mind=mind, colour=colour, surrogate_mind_type=VWUserMindSurrogate)
163        except TypeError as e:
164            VWRunner.__print_sub_error_message(message=e.args[0])
165
166            raise VWRunnerException(message="One or more mind surrogates are not of the allowed type. Please check the error message above.") from e
167        except VWSurrogateMindException as e:
168            VWRunner.__print_sub_error_message(message=e.args[0])
169
170            raise VWRunnerException(message="Could not validate one or more mind surrogates. Please check the error message above.") from e
171        except Exception as e:
172            VWRunner.__print_sub_error_message(message=e.args[0])
173
174            raise VWInternalError(message="Unknown error while validating one or more mind surrogates. Please check the error message above.") from e

194    def __validate_play_load(self) -> None:

196            raise TypeError("Argument `play` must be a boolean.")
197
198        if not isinstance(self.__args["load"], self.__allowed_args["load"]):
199            raise TypeError("Argument `load` must be a string.")

208    def __validate_speed(self) -> None:
209        if not isinstance(self.__args["speed"], self.__allowed_args["speed"]):
210            raise TypeError("Argument `speed` must be a float.")
211
212        if cast(float, self.__args["speed"]) < 0.0 or cast(float, self.__args["speed"]) >= 1:
213            raise ValueError("Argument \"speed\" must be >=0 and < 1.")
214
215    def __validate_scale(self) -> None:
216        if not isinstance(self.__args["scale"], self.__allowed_args["scale"]):
217            raise TypeError("Argument `scale` must be a float.")
218
219        # A 0 value is equivalent to omitting the argument from `vacuumworld.run()`.
220        if cast(float, self.__args["scale"]) < 0.0 or cast(float, self.__args["scale"]) > 2.5:
221            raise ValueError("Argument \"scale\" must be >= 0 and <= 2.5.")
222
223    def __validate_tooltips(self) -> None:
224        if not isinstance(self.__args["tooltips"], self.__allowed_args["tooltips"]):
225            raise TypeError("Argument `tooltips` must be a boolean.")

227    def __validate_total_cycles(self) -> None:
228        if not isinstance(self.__args["total_cycles"], self.__allowed_args["total_cycles"]):
229            raise TypeError("Argument `total_cycles` must be an integer.")
230
231        # A 0 value means an infinite number of cycles.
232        if cast(int, self.__args["total_cycles"]) < 0:
233            raise ValueError("Argument \"total_cycles\" must be >= 0.")

235    def __validate_efforts(self) -> None:
236        if not isinstance(self.__args["efforts"], dict):
237            raise TypeError(f"Invalid type for argument `efforts`: it should be `dict[str, int]`, but it is `{type(self.__args['efforts'])}`")
238        elif not VWValidator.does_type_match_for_all_elements(t=str, iterable=[key for key in self.__args["efforts"].keys()]):
239            raise TypeError("Invalid type for argument `efforts`: it should be `dict[str, int]`, but there is at least a key that is not a `str`")
240        elif not VWValidator.does_type_match_for_all_elements(t=int, iterable=[value for value in self.__args["efforts"].values()]):
241            raise TypeError("Invalid type for argument `efforts`: it should be `dict[str, int]`, but there is at least a value that is not an `int`")
242        elif not all([effort_name in VWActionEffort.EFFORTS for effort_name in self.__args["efforts"]]):
243            raise ValueError(f"Invalid effort name: it should be one of {[k for k in VWActionEffort.EFFORTS]}, but it is `{[e for e in self.__args['efforts'] if e not in VWActionEffort.EFFORTS][0]}`")
244

247            raise TypeError("Argument `debug_enabled` must be a boolean.")

249    def __override_default_config(self) -> None:

251        # The content of `self.__args` has already been validated in `__validate_optional_args()`.

253        self.__config["white_mind_filename"] = getsourcefile(self.__minds[VWColour.white].__class__)

255        self.__config["green_mind_filename"] = getsourcefile(self.__minds[VWColour.green].__class__)
256        self.__config["user_mind_filename"] = getsourcefile(self.__minds[VWColour.user].__class__)

263        self.__config["file_to_load"] = cast(str, self.__args["load"])
264        self.__config["scale"] = cast(float, self.__args["scale"])
265        self.__config["x_scale"] = cast(float, self.__args["scale"])
266        self.__config["y_scale"] = cast(float, self.__args["scale"])
267        self.__config["tooltips"] = cast(bool, self.__config["tooltips"]) and cast(bool, self.__args["tooltips"])
268        self.__config["total_cycles"] = cast(int, self.__args["total_cycles"])
269        self.__config["debug"] = cast(bool, self.__config["debug"]) and cast(bool, self.__args["debug_enabled"])

303            else:

306                VWActionEffort.override_default_effort_for_action(action_name=k, new_effort=v)

325        VWActorBehaviourDebugger.DEBUG_ENABLED = self.__config["debug"]

328            VWActorBehaviourDebugger.PRIMES = cast(list[int], self.__config["debug_primes_test"])
