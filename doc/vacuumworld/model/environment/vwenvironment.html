 39class VWEnvironment(Environment):
 40    '''
 41    This class represents the environment in which the simulation takes place.
 42
 43    A `VWEnvironment` is made of:
 44
 45    * A `VWAmbient` (itself a wrapper for a grid mapping `VWCoord` objects to `VWLocation` objects)
 46
 47    * A collection of `VWActor` objects.
 48
 49    * A collection of `VWDirt` objects.
 50
 51    An API is provided to interact with the `VWAmbient`, the `VWActor` objects, and the `VWDirt` objects.
 52
 53    A `VWEnvironment` evolves in cycles, starting at cycle `0`. Each environmental cycle is composed of:
 54
 55    * A feedback phase, in which the `VWEnvironment` sends to each `VWActor` a `VWObservation` of the surroundings of such `VWActor` and pending messages from other `VWActor` instances.
 56
 57    * A physical/communicative evolution phase, in which each `VWActor` potentially attempts to modify the `VWEnvironment` via a `VWPhysicalAction`, and potentially engages in communications via a `VWCommunicativeAction`.
 58    Only one `VWPhysicalAction` and one `VWCommunicativeAction` can be attempted per cycle per `VWActor`.
 59    '''

128    def __validate_number_of_actions(self, actions: list[Action]) -> None:
129        n: int = cast(int, self.__config["max_number_of_actions_per_actor_per_cycle"])
130
131        assert n > 0
132
133        if len(actions) == 0:
134            raise VWActionAttemptException("No actions were attempted. At least 1 action must be attempted per cycle.")
135        elif len(actions) > n:
136            raise VWActionAttemptException(f"Too many actions were attempted. There is a hard limit of {n} actions per actor per cycle.")

138    @staticmethod
139    def __validate_action_types(actions: list[Action]) -> None:
140        for action in actions:
141            if not isinstance(action, VWPhysicalAction) and not isinstance(action, VWCommunicativeAction):
142                raise VWMalformedActionException(f"Unrecognised action: {type(action)}.")
143
144    def __validate_pool_of_actions(self, actions: list[Action]) -> None:
145        n: int = cast(int, self.__config["max_number_of_actions_per_actor_per_cycle"])
146        n_physical: int = cast(int, self.__config["max_number_of_physical_actions_per_actor_per_cycle"])
147        n_communicative: int = cast(int, self.__config["max_number_of_communicative_actions_per_actor_per_cycle"])
148
149        # Self-consistency assertion in the config data.
150        assert n_physical + n_communicative == n
151
152        # The check that at least one `VWAction` (and no more than `n`) is attempted has already been done in `__validate_number_of_actions()`. Hence, this is an assertion, and not a conditional statement.
153        assert 0 < len(actions) <= n
154
155        if sum(map(lambda action: isinstance(action, VWPhysicalAction), actions)) > n_physical:
156            raise VWActionAttemptException(f"Too many physical actions were attempted. There is a hard limit of {n_physical} physical action, and {n_communicative} communicative action per actor per cycle.")
157
158        if sum(map(lambda action: isinstance(action, VWCommunicativeAction), actions)) > n_communicative:
159            raise VWActionAttemptException(f"Too many communicative actions were attempted. There is a hard limit of {n_physical} physical action, and {n_communicative} communicative action per actor per cycle.")

355    def from_json(data: dict[str, JSONValue], config: dict[str, JSONValue]) -> VWEnvironment:
356        '''
357        Creates and returns a `VWEnvironment` from the specified JSON representation (`data`) and `config`.
358
359        Each `VWActor` will have a fresh ID and progressive ID, each `VWUser` will be in whatever mode is the default one.
360        '''
361        grid: dict[VWCoord, VWLocation] = {}
362        actors: list[VWActor] = []
363        dirts: list[VWDirt] = []
364
365        if not data:
366            return VWEnvironment.generate_empty_env(config=config)
367
368        assert "locations" in data and data["locations"] is not None and isinstance(data["locations"], list)
369
370        for location_data in data["locations"]:
371            assert isinstance(location_data, dict)
372
373            coord_data: dict[str, int] = cast(dict[str, int], location_data["coords"])
374            coord: VWCoord = VWCoord(x=coord_data["x"], y=coord_data["y"])
375            actor, actor_appearance = VWEnvironment.__load_actor(location_data=location_data)
376            dirt, dirt_appearance = VWEnvironment.__load_dirt(location_data=location_data)
377
378            assert actor and actor_appearance or not actor and not actor_appearance
379            assert dirt and dirt_appearance or not dirt and not dirt_appearance
380
381            actors = actors + [actor.or_else_raise()] if actor.is_present() else actors
382            dirts = dirts + [dirt.or_else_raise()] if dirt.is_present() else dirts
383            wall_data: dict[str, bool] = cast(dict[str, bool], location_data["wall"])
384            wall: dict[VWOrientation, bool] = {o: wall_data[str(o)] for o in VWOrientation}
385
386            grid[coord] = VWLocation(coord=coord, actor_appearance=actor_appearance, dirt_appearance=dirt_appearance, wall=wall)
387
388        VWEnvironment.__validate_grid(grid=grid, config=config)
389
390        return VWEnvironment(config=config, ambient=VWAmbient(grid=grid), initial_actors=actors, initial_dirts=dirts)

458    @staticmethod
459    def __validate_grid(grid: dict[VWCoord, VWLocation], config: dict[str, JSONValue], candidate_grid_line_dim: int=-1) -> None:
460        assert grid is not None and isinstance(grid, dict)
461
462        tmp: float = sqrt(len(grid))
463        grid_line_dim: int = floor(tmp)
464
465        assert grid_line_dim == tmp
466

 42    This class represents the environment in which the simulation takes place.
 43
 44    A `VWEnvironment` is made of:
 45
 46    * A `VWAmbient` (itself a wrapper for a grid mapping `VWCoord` objects to `VWLocation` objects)
 47
 48    * A collection of `VWActor` objects.
 49
 50    * A collection of `VWDirt` objects.

 53
 54    A `VWEnvironment` evolves in cycles, starting at cycle `0`. Each environmental cycle is composed of:
 55
 56    * A feedback phase, in which the `VWEnvironment` sends to each `VWActor` a `VWObservation` of the surroundings of such `VWActor` and pending messages from other `VWActor` instances.

 58    * A physical/communicative evolution phase, in which each `VWActor` potentially attempts to modify the `VWEnvironment` via a `VWPhysicalAction`, and potentially engages in communications via a `VWCommunicativeAction`.

357        '''
358        Creates and returns a `VWEnvironment` from the specified JSON representation (`data`) and `config`.
359
360        Each `VWActor` will have a fresh ID and progressive ID, each `VWUser` will be in whatever mode is the default one.
361        '''
362        grid: dict[VWCoord, VWLocation] = {}
363        actors: list[VWActor] = []
364        dirts: list[VWDirt] = []
365
366        if not data:
367            return VWEnvironment.generate_empty_env(config=config)
368
369        assert "locations" in data and data["locations"] is not None and isinstance(data["locations"], list)
370
371        for location_data in data["locations"]:
372            assert isinstance(location_data, dict)
373
374            coord_data: dict[str, int] = cast(dict[str, int], location_data["coords"])

376            actor, actor_appearance = VWEnvironment.__load_actor(location_data=location_data)

379            assert actor and actor_appearance or not actor and not actor_appearance
380            assert dirt and dirt_appearance or not dirt and not dirt_appearance

382            actors = actors + [actor.or_else_raise()] if actor.is_present() else actors
383            dirts = dirts + [dirt.or_else_raise()] if dirt.is_present() else dirts

385            wall: dict[VWOrientation, bool] = {o: wall_data[str(o)] for o in VWOrientation}

This class represents the environment in which the simulation takes place.

A VWEnvironment is made of:

  • A VWAmbient (itself a wrapper for a grid mapping VWCoord objects to VWLocation objects)

  • A collection of VWActor objects.

  • A collection of VWDirt objects.

An API is provided to interact with the VWAmbient, the VWActor objects, and the VWDirt objects.

A VWEnvironment evolves in cycles, starting at cycle 0. Each environmental cycle is composed of:

  • A feedback phase, in which the VWEnvironment sends to each VWActor a VWObservation of the surroundings of such VWActor and pending messages from other VWActor instances.

  • A physical/communicative evolution phase, in which each VWActor potentially attempts to modify the VWEnvironment via a VWPhysicalAction, and potentially engages in communications via a VWCommunicativeAction. Only one VWPhysicalAction and one VWCommunicativeAction can be attempted per cycle per VWActor.
